add_library(ScriptVessel SHARED
	ScriptVessel.cpp
)

set_target_properties(ScriptVessel
	PROPERTIES
	FOLDER Vessels
)

	PUBLIC ${ORBITER_SOURCE_SDK_INCLUDE_DIR}

	${SpaceXpanseTgt}
	SpaceXpansesdk
)

	RUNTIME
	DESTINATION ${ORBITER_INSTALL_MODULE_DIR}
)
